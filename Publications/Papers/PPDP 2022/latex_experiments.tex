\paragraph{Example of proof tree}

$
\infer[(E\wedge_1)]{\Gamma \vdash A}{
  \infer{\Gamma \vdash A \wedge B}{\vdash A & \vdash B}
  }
\qquad
\infer[(E\wedge)]{\Gamma \vdash B}{\Gamma \vdash A \wedge B}
$
